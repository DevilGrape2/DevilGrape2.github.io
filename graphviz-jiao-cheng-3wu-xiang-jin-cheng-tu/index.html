graphviz教程3（无向进程图）

本次博文的内容主要为：使用python代码画一个进程图。如下为代码实现的过程。

教程如下

from graphviz import Graph

g = Graph('G', filename='process.gv', engine='sfdp',format='png')
#g.attr(dpi='500')
g.edge('run', 'intr')
g.edge('intr', 'runbl')
g.edge('runbl', 'run')
g.edge('run', 'kernel')
g.edge('kernel', 'zombie')
g.edge('kernel', 'sleep')
g.edge('kernel', 'runmem')
g.edge('sleep', 'swap')
g.edge('swap', 'runswap')
g.edge('runswap', 'new')
g.edge('runswap', 'runmem')
g.edge('new', 'runmem')
g.edge('sleep', 'runmem')
g.render(r'C:\Users\just\Desktop\进程图')
g.view()
